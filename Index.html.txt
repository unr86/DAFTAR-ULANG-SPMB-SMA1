  <script>
    function prosesCari() {
      var nis = document.getElementById("nisInput").value;
      if (!nis) {
        alert("Masukkan NIS terlebih dahulu!");
        return;
      }

      document.getElementById("loader").style.display = "block";
      document.getElementById("hasilCard").style.display = "none";

      google.script.run
        .withSuccessHandler(tampilkanHasil)
        .withFailureHandler(function(err) {
            alert("Error: " + err);
            document.getElementById("loader").style.display = "none";
         })
        .cariSiswa(nis);
    }

    function tampilkanHasil(hasil) {
      document.getElementById("loader").style.display = "none";
      
      if (hasil.status === "tidak_ditemukan") {
        alert("Maaf, NIS tidak ditemukan. Silakan hubungi panitia.");
        return;
      }

      document.getElementById("resNama").innerText = hasil.nama;
      document.getElementById("resAsal").innerText = hasil.asalSekolah;
      document.getElementById("resJalur").innerText = hasil.jalur;
      
      var ketBadge = document.getElementById("resKeterangan");
      ketBadge.innerText = hasil.keterangan;
      
      // Pewarnaan status
      if (hasil.keterangan.toLowerCase().includes("lulus") || hasil.keterangan.toLowerCase().includes("selesai")) {
        ketBadge.className = "badge-status bg-success text-white";
      } else {
        ketBadge.className = "badge-status bg-warning text-dark";
      }

      var btnDownload = document.getElementById("btnDownload");
      if (hasil.linkBukti) {
        btnDownload.href = hasil.linkBukti;
        btnDownload.style.display = "block";
      } else {
        btnDownload.style.display = "none";
      }

      document.getElementById("hasilCard").style.display = "block";
    }
  </script>
  <script>
  function updateWaktu() {
    var sekarang = new Date();
    
    // Opsi pengaturan format tanggal Indonesia
    var opsiTanggal = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
    var tanggalIndo = sekarang.toLocaleDateString('id-ID', opsiTanggal);
    
    // Format waktu (Jam:Menit:Detik)
    var jam = String(sekarang.getHours()).padStart(2, '0');
    var menit = String(sekarang.getMinutes()).padStart(2, '0');
    var detik = String(sekarang.getSeconds()).padStart(2, '0');
    var waktuIndo = jam + ':' + menit + ':' + detik + ' WIB';
    
    // Masukkan ke elemen HTML
    document.getElementById('infoTanggal').innerText = tanggalIndo;
    document.getElementById('infoWaktu').innerText = waktuIndo;
  }

  // Jalankan fungsi pertama kali saat halaman dibuka
  updateWaktu();
  // Perbarui jam setiap 1 detik (1000 milidetik)
  setInterval(updateWaktu, 1000);
</script>
